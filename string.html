toLowerCase

public String toLowerCase() { String result = ""; for (int i = 0; i < str.length(); i++) { char ch = str.charAt(i); if (ch >= 'A' && ch <= 'Z') { result += (char)(ch + 32); } else { result += ch; } } return result; }
new MyString("HELLO").toLowerCase() → "hello"

toUpperCase

public String toUpperCase() { String result = ""; for (int i = 0; i < str.length(); i++) { char ch = str.charAt(i); if (ch >= 'a' && ch <= 'z') { result += (char)(ch - 32); } else { result += ch; } } return result; }
new MyString("hello").toUpperCase() → "HELLO"

split

public String[] split(char delimiter) { java.util.List parts = new java.util.ArrayList<>(); String current = ""; for (int i = 0; i < str.length(); i++) { if (str.charAt(i) == delimiter) { parts.add(current); current = ""; } else { current += str.charAt(i); } } parts.add(current); return parts.toArray(new String[0]); }
new MyString("a-b-c").split('-') → ["a", "b", "c"]
